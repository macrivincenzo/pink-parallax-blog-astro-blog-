def generate_haro_response(query):
    # Use ChatGPT to generate response
    response = openai.ChatCompletion.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system", "content": "You are a business valuation expert"},
            {"role": "user", "content": f"Write a HARO response for: {query}"}
        ]
    )
    
    return response.choices[0].message.content

def send_haro_response(query, journalist_email):
    # Generate response
    response = generate_haro_response(query)
    
    # Send email
